Index:      1     2     3     4     5     6     7
Value:     50    30    40    10    20    35    25

            50
          /    \
        30      40
       /  \    /  \
     10   20  35  25

public class Min_Heap {

    // Moves the element at index `i` down the heap until the heap property is restored.
    public static void Sink(int[] arr, int i, int n) {
        int smallest = i;
        int left = 2 * i;
        int right = 2 * i + 1;

        // Compare with left child
        if (left <= n && arr[left] < arr[smallest]) {
            smallest = left;
        }

        // Compare with right child
        if (right <= n && arr[right] < arr[smallest]) {
            smallest = right;
        }

        // If the smallest is not the current node, swap and continue sinking
        if (smallest != i) {
            int temp = arr[i];
            arr[i] = arr[smallest];
            arr[smallest] = temp;
            Sink(arr, smallest, n);
        }
    }

    // Moves the element at index `i` up the heap until the heap property is restored.
    public static void Swim(int[] arr, int i) {
        while (i > 1 && arr[i / 2] > arr[i]) {
            int temp = arr[i];
            arr[i] = arr[i / 2];
            arr[i / 2] = temp;
            i = i / 2;
        }
    }

    // Converts an unordered array into a valid min heap.
    public void Build_Min_Heap(int[] arr, int n) {
        for (int i = n / 2; i >= 1; i--) {
            Sink(arr, i, n);
        }
    }

    // Performs heap sort (ascending order).
    public void Heap_Sort(int[] arr) {
        int n = arr.length - 1;
        Build_Min_Heap(arr, n);
        for (int i = n; i >= 2; i--) {
            int temp = arr[1];
            arr[1] = arr[i];
            arr[i] = temp;
            Sink(arr, 1, i - 1);
        }
    }

    // Inserts a new key into the heap.
    public int insert(int[] arr, int n, int key) {
        if (n + 1 >= arr.length) {
            System.out.println("Heap is full");
            return n;
        }
        n++;
        arr[n] = key;
        Swim(arr, n);
        return n;
    }

    // Deletes the minimum element (root) from the heap.
    public int delete(int[] arr, int n) {
        if (n < 1) {
            System.out.println("Heap is empty");
            return n;
        }
        arr[1] = arr[n];
        n--;
        Sink(arr, 1, n);
        return n;
    }

    // Returns the minimum value (at the root) without removing it.
    public int peek(int[] arr) {
        int n = arr.length - 1;
        if (n < 1) {
            throw new RuntimeException("Heap is empty");
        }
        return arr[1];
    }

    // Increases the value of an element (may move it down)
    public void increase_key_Min_Heap(int[] arr, int i, int new_value) {
        if (new_value < arr[i]) {
            System.out.println("New value is smaller than current value");
            return;
        }
        arr[i] = new_value;
        Sink(arr, i, arr.length - 1);
    }

    // Decreases the value of an element (may move it up)
    public void decrease_key_Min_Heap(int[] arr, int i, int new_value) {
        if (new_value > arr[i]) {
            System.out.println("New value is larger than current value");
            return;
        }
        arr[i] = new_value;
        Swim(arr, i);
    }

    // Checks whether the heap is empty (n = 0).
    public boolean isEmpty(int n) {
        return n == 0;
    }

    // Removes and returns the minimum element (root).
    public int extract_min(int[] arr, int n) {
        if (isEmpty(n)) {
            System.out.println("Heap is empty");
            return -1;
        }
        int min = arr[1];
        arr[1] = arr[n];
        n--;
        Sink(arr, 1, n);
        return min;
    }

    // Returns the maximum element in the heap without removing it (scans all elements).
    public int extract_max(int[] arr, int n) {
        if (isEmpty(n)) {
            System.out.println("Heap is empty");
            return -1;
        }
        int max = arr[1];
        for (int i = 2; i <= n; i++) {
            if (arr[i] > max) {
                max = arr[i];
            }
        }
        return max;
    }
}      


public class Min_Heap_Main {
    public static void main(String[] args) {
        int[] heap = new int[100];  // 1-based indexing
        int n = 0;
        Min_Heap minHeap = new Min_Heap();

        n = minHeap.insert(heap, n, 20);
        n = minHeap.insert(heap, n, 15);
        n = minHeap.insert(heap, n, 30);
        System.out.println("Min: " + minHeap.peek(heap));
        n = minHeap.delete(heap, n);
        System.out.println("After delete, new Min: " + minHeap.peek(heap));
    }
}
